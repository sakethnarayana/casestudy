// Reference to the table body var tableBody = document.querySelector("#myTable tbody"); // Loop through the JSON data and populate the table jsonData.forEach(function (data) { // Create a new row var row = tableBody.insertRow(); // Add a checkbox in the first cell of each row var checkboxCell = row.insertCell(0); var checkbox = document.createElement("input"); checkbox.type = "checkbox"; checkboxCell.appendChild(checkbox); // Add other cells with data Object.keys(data).forEach(function (key, index) { var cell = row.insertCell(index + 1); // Loop through checkboxes and check if they are checked checkboxes.forEach(function (checkbox, index) { if (checkbox.checked) { // Get the corresponding ID from the data var id = jsonData[index].id; selectedIds.push(id); } }); // Display the selected IDs alert("Selected IDs: " + selectedIds.join(', ')); } </script> </body> </html> </plaintext> </body> </html>
